📦 Installation

1

Install via Composer

composer require lovegem/framework
2

Generate Application Key

php artisan key:generate
3

Start Development Server

php artisan serve

🚀 Quick Start

Create Your First Route

<?php use LoveGem\Support\Facades\Route; Route::get('/', function () { return 'Hello from LoveGem!'; });

Create a Model

php artisan make:model Post
<?php namespace App\Models; use LoveGem\Database\Eloquent\Model; class Post extends Model { protected $fillable = ['title', 'content']; }

Use Eloquent

// Get all posts $posts = Post::all(); // Create a post Post::create([ 'title' => 'My First Post', 'content' => 'Hello World!', ]); // Find by ID $post = Post::find(1); // Where clause $posts = Post::where('active', true)->get();
